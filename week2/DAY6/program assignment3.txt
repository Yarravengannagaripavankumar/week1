            <label>Password:</label><br>
            <input type="password" id="password"><br><br>
        </div>

        <div>
            <button type="submit">Submit</button>
        </div>

<script>
    const form = document.getElementById("loginForm");

    form.addEventListener("submit", function (event) {
        event.preventDefault();

        const username = document.getElementById("username").value;
        const password = document.getElementById("password").value;

        const usernamePattern = /^[A-Za-z]{5,}$/;
        const passwordPattern = /^(?=.*[A-Z])(?=.*[0-9])(?=.*[^A-Za-z0-9]).{8,}$/;

        if (username === "" && password === "") {
            alert("Username and Password cannot be empty");
        } 
        else if (!usernamePattern.test(username)) {
            alert("Username must be at least 5 letters and contain only alphabets");
        } 
        else if (!passwordPattern.test(password)) {
            alert("Password must be at least 8 characters and include 1 capital letter, 1 number, and 1 special character");
        } 
        else {
            alert("Login successful");
        }
    });
</script>
