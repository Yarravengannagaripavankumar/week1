7. Write a Script to print all the Prime numbers between $1 and $2
#!/bin/sh

from=$1
to=$2

for num in $(seq $from $to)
do
    if [ $num -le 1 ]; then
        continue
    fi

    prime=1

    for i in $(seq 2 $((num - 1)))
    do
        if [ $((num % i)) -eq 0 ]; then
            prime=0
            break
        fi
    done

    if [ $prime -eq 1 ]; then
        echo $num
    fi
done

count=0

for num in $(seq 2 1000)
do
    prime=1

    for i in $(seq 2 $((num - 1)))
    do
        if [ $((num % i)) -eq 0 ]; then
            prime=0
            break
        fi
    done

    if [ $prime -eq 1 ]; then
        count=$((count + 1))
    fi

    if [ $count -eq $n ]; then
        echo $num
        break
    fi
done

9. Find the 2nd smallest digit of a number
#!/bin/bash

n=$1
smallest=10
sec_smallest=10

while [ $n -gt 0 ]
do
    digit=$((n % 10))

    if [ $digit -lt $smallest ]; then
        sec_smallest=$smallest
        smallest=$digit
    elif [ $digit -ne $smallest ] && [ $digit -lt $sec_smallest ]; then
        sec_smallest=$digit
    fi

    n=$((n / 10))
done

echo "$sec_smallest"

12. Given a number, print the number which is the very next possible number having all the digits with same frequency as that of the i/p number.
#!/bin/bash
#
read -p "Enter number: " n
next=$((n + 1))

while true
do
    if [ "$(echo $n | fold -w1 | sort | tr -d '\n')" = \
         "$(echo $next | fold -w1 | sort | tr -d '\n')" ]
    then
        echo "Next number: $next"
        break
    fi
    next=$((next + 1))
done
