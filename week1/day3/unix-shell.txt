3. Program to Check if User-Given Number is Even
#!/bin/bash
echo "Enter a number to check if it is Even:"
read inputNumber
result=$(expr $inputNumber % 2)

if [ $result -eq 0 ]
then
    echo "$inputNumber is Even"
else
    echo "$inputNumber is Odd"
fi

4. Alternate Even/Odd Program Using test
#!/bin/bash
echo "Enter a number:"
read num
result=$(expr $num % 2)
if test $result -eq 0
then
    echo "Number is Even"
else
    echo "Number is Odd"
fi

5. Leap Year Program (Using read & Logical Operators
#!/bin/bash
#
echo "Enter the year to check if it is Leap year:"
read inputYear

if [ $((inputYear % 4)) -eq 0 ] && [ $((inputYear % 100)) -ne 0 ] || [ $((inputYear % 400)) -eq 0 ]
then
    echo "$inputYear is a Leap Year"
else
    echo "$inputYear is not a Leap Year"
fi

6. Leap Year Program Using Command-Line Argument ($1)
#!/bin/bash

year=$1

if [ -z "$year" ]; then
    echo "Usage: $0 <year>"
    exit 1
fi

if (( (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0) )); then
    echo "$year is a leap year"
else
    echo "$year is not a leap year"
fi
How to Run
sh leap.sh 2024

7.Factorial Program Using while Loop
Script: factorial.sh
#!/bin/bash

inputNumber=$1
factorialNumber=1
tempNumber=$inputNumber

while [ $tempNumber -ge 1 ]
do
    factorialNumber=$((factorialNumber * tempNumber))
    tempNumber=$((tempNumber - 1))
done

echo "Factorial of $inputNumber is $factorialNumber"
exit 0
How to Run
sh factorial.sh 5

8. Prime Numbers Between Two Numbers
Script: prime.sh
#!/bin/bash

fromNumber=$1
toNumber=$2
tempNumber=$fromNumber

while [ $tempNumber -le $toNumber ]
do
    prime=1
    i=2

    while [ $i -lt $tempNumber ]
    do
        if [ $((tempNumber % i)) -eq 0 ]
        then
            prime=0
            break
        fi
        i=$((i + 1))
    done

    if [ $prime -eq 1 ] && [ $tempNumber -gt 1 ]
    then
        echo $tempNumber
    fi

    tempNumber=$((tempNumber + 1))
done

exit 0
